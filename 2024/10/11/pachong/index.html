import pandas as pd
import matplotlib.pyplot as plt
plt.rcParams[‘font.sans-serif’]=[‘SimHei’]
plt.rcParams[‘axes.unicode_minus’]=False

#读取 Excel 文件
df = pd.read_excel(‘豆瓣排行榜爱情.xlsx’)

region_counts = {} # 初始化一个空字典来存储地区及其计数
for _, row in df.iterrows(): # 遍历df中的每一行
# 处理regions字段，分割字符串并清洗数据
regions = [item.strip().replace(“‘“, ‘’) for item in row[‘regions’][1:-1].split(“‘, ‘“)]
for region in regions: # 地区计数
if region not in region_counts:
region_counts[region] = 0
region_counts[region] += 1
sizes = list(region_counts.values()) # 地区计数
labels = list(region_counts.keys()) # 地区名称
plt.figure(figsize=(12, 8)) # 设置图形大小
plt.pie(sizes, labels=labels, autopct=’%1.1f%%’, startangle=140, textprops={‘fontsize’: 10})

plt.title(‘各地区电影数量占比’) # 设置图形标题

#优化饼图的显示效果
plt.axis(‘equal’) # 保持饼图为圆形

country_scores = {} # 初始化一个空字典来存储每个国家的所有评分
for index, movie in df.iterrows(): # 遍历df中的每一行
# 解析regions字段，分割字符串
regions = [region.strip().replace(“‘“, ‘’) for region in movie[‘regions’][1:-1].split(“,”)]
score = float(movie[‘score’]) # 获取评分并转换为浮点数
for region in regions: # 检查地区是否为目标国家之一，并累加评分
if region in country_scores:
country_scores[region].append(score)
else:
country_scores[region] = [score]
#计算每个国家的平均评分
country_avg_scores = {country: sum(scores) / len(scores) for country, scores in country_scores.items()}
countries = list(country_avg_scores.keys()) # 国家列表
avg_scores = list(country_avg_scores.values()) # 对应的平均评分列表

#绘制柱状图
plt.figure(figsize=(10, 8)) # 设置图形大小
plt.bar(countries, avg_scores, color=’skyblue’)

#设置X轴和Y轴的标签
plt.xlabel(‘地区’)
plt.ylabel(‘平均评分’)
plt.title(‘各地区电影平均评分’)

#显示数值标签在每个柱子上
for i, score in enumerate(avg_scores):
plt.text(countries[i], score, f’{score:.2f}’, ha=’center’, va=’bottom’)

#优化显示效果
plt.xticks(rotation=45) # 旋转X轴标签，防止重叠
plt.tight_layout() # 自动调整布局以确保内容显示完整

#显示图形
plt.show()
